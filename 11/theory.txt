Създаване (отваряне) на семафор
#include <semaphore.h>
#include <sys/stat.h>
#include <fcntl.h>
sem_t *sem_open(const char *name, int oflag[, mode_t mode, unsigned int value]) - създава и инициализира (отваря) семафор

oflag:
O_CREAT - създава семафор
O_EXCL - заедно с O_CREAT, ако семафора съществува връща грешка EEXIST

Увеличаване на стойността (деблокиране)
#include <semaphore.h>
int sem_post(sem_t *sem) - уваличава стойността на семафора с 1

Намаляване на стойността (блокиране)
#include <semaphore.h>
int sem_wait(sem_t *sem) - намалява стойността на семафора с 1
int sem_trywait(sem_t *sem) - намалява стойността на семафора с 1; не се блокира, а връща грешка EAGAIN

Получаване на стойността
#include <semaphore.h>
int sem_getvalue(sem_t *sem, int *sval) - взема стойността на семафор

Затваряне на семафор
#include <semaphore.h>
int sem_close(sem_t *sem) - затваря семафора

Унищожаване на семафор
#include <semaphore.h>
int sem_unlink(const char *name) - унищожава семафора
